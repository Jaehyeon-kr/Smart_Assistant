## 🎯 핵심 구현 예제

### 예제 1: 간단한 회원가입

**프론트엔드 (RegisterPage.jsx)**
```javascript
import { useState } from 'react';
import api from '../api/api';

export default function RegisterPage() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [error, setError] = useState('');

  const handleRegister = async (e) => {
    e.preventDefault();
    try {
      await api.post('/auth/register', { email, password });
      alert('회원가입 성공! 로그인하세요.');
      window.location.href = '/login';
    } catch (err) {
      setError(err.response?.data?.error || '회원가입 실패');
    }
  };

  return (
    <form onSubmit={handleRegister}>
      <input
        type="email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        placeholder="이메일"
      />
      <input
        type="password"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
        placeholder="비밀번호"
      />
      <button type="submit">회원가입</button>
      {error && <p style={{ color: 'red' }}>{error}</p>}
    </form>
  );
}
```

**백엔드 (routes/auth.js)**
```javascript
const express = require('express');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const router = express.Router();

// 회원가입
router.post('/register', async (req, res) => {
  const { email, password } = req.body;

  // 입력 검증 (생략)

  try {
    // bcrypt로 해싱
    const hash = await bcrypt.hash(password, 10);

    // DB 저장 (sqlite3 사용)
    // 성공 응답
    res.json({ data: { userId: 1, email } });
  } catch (err) {
    res.status(500).json({ error: '회원가입 실패' });
  }
});

module.exports = router;
```

### 예제 2: PDF 업로드

**프론트엔드**
```javascript
import api from '../api/api';

const handlePDFUpload = async (file) => {
  const formData = new FormData();
  formData.append('file', file);

  try {
    const res = await api.post('/pdfs', formData, {
      headers: { 'Content-Type': 'multipart/form-data' }

    console.log('업로드 성공:', res.data);
    // /documents 새로고침
  } catch (err) {
    console.error('업로드 실패:', err);
  }
};
```

**백엔드**
```javascript
const multer = require('multer');
const pdf = require('pdf-parse');

const upload = multer({ dest: './uploads' });

router.post('/pdfs', upload.single('file'), async (req, res) => {
  try {
    // pdf-parse로 텍스트 추출
    const dataBuffer = fs.readFileSync(req.file.path);
    const data = await pdf(dataBuffer);

    const extractedText = data.text; // 추출된 텍스트

    // DB 저장 + Claude API 호출 (Promise.all)
    // 응답: { pdfId, originalName, summaryStatus, questionsStatus }

    res.json({
      data: {
        pdfId: 1,
        originalName: req.file.originalname,
        summaryStatus: 'processing',
        questionsStatus: 'processing'
      }
    });
  } catch (err) {
    res.status(500).json({ error: 'PDF 업로드 실패' });
  }

```

### 예제 3: AI 튜터 (SSE)

**프론트엔드**
```javascript
const handleChat = (message) => {
  const eventSource = new EventSource(
    `/api/pdfs/${pdfId}/chat?message=${encodeURIComponent(message)}`
  );

  eventSource.onmessage = (event) => {
    const data = JSON.parse(event.data);
    if (data.chunk === '[DONE]') {
      eventSource.close();
    } else {
      setResponse(prev => prev + data.chunk);
    }
  };

  eventSource.onerror = () => {
    setError('응답 실패. 다시 시도하세요.');
    eventSource.close();
  };
};
```

**백엔드**
```javascript
const { Anthropic } = require('@anthropic-ai/sdk');

router.get('/pdfs/:pdfId/chat', async (req, res) => {
  const { message } = req.query;
  const { pdfId } = req.params;

  // PDF 콘텍스트 조회
  const pdf = await db.get('SELECT extractedText FROM pdfs WHERE pdfId = ?', [pdfId]);

  const client = new Anthropic();

  res.setHeader('Content-Type', 'text/event-stream');

  const stream = client.messages.stream({
    model: 'claude-sonnet-4-20250514',
    max_tokens: 1024,
    messages: [
      {
        role: 'user',
        content: `PDF 내용: ${pdf.extractedText}\n\n질문: ${message}`
      }
    ]
  });

  stream.on('text', (text) => {
    res.write(`data: ${JSON.stringify({ chunk: text })}\n\n`);
  });

  stream.finalMessage().then(() => {
    res.write('data: {"chunk":"[DONE]"}\n\n');
    res.end();
  });
});
```


## 📞 트러블슈팅

### Q: "CORS 오류"
A: `backend/server.js`에서 CORS 설정 확인
```javascript
app.use(cors({
  origin: 'http://localhost:3000',
  credentials: true
}));
```

### Q: "JWT 만료"
A: localStorage에서 토큰 확인 또는 로그인 페이지로 리다이렉트

### Q: "PDF 텍스트가 공백"
A: OCR 미지원. 텍스트 레이어가 있는 PDF만 가능

### Q: "Claude API 비용이 너무 많음"
A: Rate limiting 설정 또는 더 작은 모델 사용 고려
